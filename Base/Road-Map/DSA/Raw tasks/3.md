## Решение 1

[[Метод скользящего окна]]

```typescript
const lengthOfLongestSubstring = (s: string): number => {
  const hashMap = {};

  let leftPointer = 0;
  let rightPointer = 1;
  let currentLength = 1;
  let maxLength = 1;

  hashMap[s[leftPointer]] = true;

  for (;rightPointer < s.length;) {
    if (hashMap[s[rightPointer]] !== undefined) {
      delete hashMap[s[leftPointer]];

      leftPointer += 1;
      currentLength -= 1;
    } else {
      hashMap[s[rightPointer]] = true;

      rightPointer += 1;
      currentLength += 1;
    }

    maxLength = Math.max(maxLength, currentLength);
  }

  return s.length === 0 ? 0 : maxLength;
};
```

## Решение 2

[[Метод скользящего окна]]

```typescript
const lengthOfLongestSubstring = (s: string): number => {
  const set = new Set();

  let leftPointer = 0;
  let rightPointer = 1;
  let maxLength = 1;

  set.add(s[leftPointer]);

  for (;rightPointer < s.length;) {
    if (set.has(s[rightPointer])) {
      set.delete(s[leftPointer]);

      leftPointer += 1;
    } else {
      set.add(s[rightPointer]);

      rightPointer += 1;
    }

    maxLength = Math.max(maxLength, set.size);
  }

  return s.length === 0 ? 0 : maxLength;
};
```
