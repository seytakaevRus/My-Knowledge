const firstMissingPositive = (nums: number[]): number => {
  const hashMap = {};

  for (let i = 0; i < nums.length; i += 1) {
    if (nums[i] > 0) {
      hashMap[nums[i]] = nums[i];
    }
  }

  for (let i = 1; true; i += 1) {
    if (hashMap[i] === undefined) {
      return i;
    }
  }

};
