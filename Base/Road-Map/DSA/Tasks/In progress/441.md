## Решение 1

```typescript
const arrangeCoins = (n: number): number => {
  let prefix = 0;
  let sum = 0;
  let count = 0;

  for (;;) {
    count += 1;
    prefix += 1;
    sum += prefix;

    if (sum >= n) {
      return sum === n ? count : count - 1;
    }
  }
};
```

## Решение 2

```typescript
const getCoinsCountInStair = (n: number) => {
  return (n * (n + 1)) / 2;
}

const arrangeCoins = (n: number): number => {
  let left = 0;
  let right = n;
  let index = 1;

  for (; left <= right ;) {
    const middle = Math.floor((left + right) / 2);
    const countCoins = getCoinsCountInStair(middle)

    if (countCoins >= n) {
      index = middle;
    }

    if (countCoins >= n) {
      right = middle - 1;
    } else {
      left = middle + 1;
    }
  }

  const coins = getCoinsCountInStair(index)

  return coins === n ? index : index - 1;
};
```

Сложность по времени: `O()`.

Сложность по памяти: `O()`.
