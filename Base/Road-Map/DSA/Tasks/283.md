## Решение 1

```typescript
const moveZeroes = (nums: number[]): void => {
  let countOfZeros = 0;
  let pointer = 0;

  nums.forEach((value) => {
    if (value === 0) {
      countOfZeros += 1;
    } else {
      nums[pointer] = value;
      pointer += 1;
    }
  });


  for (let i = 0; i < countOfZeros; i += 1) {
    nums[nums.length - i - 1] = 0;
  }
};
```
